apres l import depuis git

composer install

php bin/console doctrine:database:create

php bin/console doctrine:migrations:migrate

php bin/console doctrine:fixtures:load

symfony server:start
symfony server:stop


php bin/console cache:clear


php bin/console make:controller  # Crée un nouveau contrôleur
php bin/console make:entity      # Crée une entité Doctrine
php bin/console make:migration   # Crée une migration vide
php bin/console make:form        # Crée une classe de formulaire
php bin/console make:user        # Crée une entité User avec système de sécurité
php bin/console make:auth        # Crée un système de login
php bin/console make:crud        # Génère un CRUD complet pour une entité
